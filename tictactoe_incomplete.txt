void TicTac::giveWinner(){ // winner message
    int result = 0;
    if (result == 1) {
        cout << "Player 1 wins!" << endl;
    }
    else if (result == 2) {
        cout << "Player 2 wins!" << endl;
    }
    else {
      cout << "It's a tie!" << endl;
    }
}
